```typescript
/**
 * Default Inertia page for `arqel::dashboard`.
 *
 * Mirrors the role of ArqelIndexPage but for dashboards. Reads
 * `props.dashboard` (server payload from `Dashboard::resolve()`)
 * and `props.filterValues` (request-time filter inputs), then
 * defers to the presentational components in
 * `@arqel-dev/ui/widgets`.
 *
 * Filter changes navigate via Inertia router.get with
 * `?filters[name]=value`, matching the contract that
 * `DashboardController::index` already reads via
 * `$request->input('filters', [])`.
 */

import type { JSX } from 'react';
import { router, usePage } from '@inertiajs/react';
import {
  DashboardFilters,
  DashboardGrid,
  WidgetRenderer,
} from '@arqel-dev/ui/widgets';

interface DashboardWidget {
  id: string;
  name: string;
  type: string;
  component: string;
  heading: string | null;
  description: string | null;
  sort: number | null;
  columnSpan: number | string;
  pollingInterval: number | null;
  deferred: boolean;
  filters: Record<string, unknown>;
  data: unknown | null;
}

interface DashboardPayload {
  id: string;
  label: string;
  path: string | null;
  widgets: DashboardWidget[];
  filters: Record<string, unknown>;
  columns: number | Record<string, number>;
  heading: string | null;
  description: string | null;
}

interface PageProps extends Record<string, unknown> {
  dashboard: DashboardPayload;
  filterValues: Record<string, unknown>;
}

export default function ArqelDashboardPage(): JSX.Element {
  const { props, url } = usePage<PageProps>();
  const { dashboard, filterValues } = props;

  const handleFilterChange = (name: string, value: unknown): void => {
    const path = typeof window !== 'undefined' ? window.location.pathname : url.split('?')[0];
    const next = { ...filterValues };
    if (value === undefined || value === null || value === '') {
      delete next[name];
    } else {
      next[name] = value;
    }
    const data: Record<string, unknown> =
      Object.keys(next).length > 0 ? { filters: next } : {};
    router.get(path ?? '/', data as Record<string, string | number | boolean>, {
      preserveState: true,
      preserveScroll: true,
      replace: true,
    });
  };

  return (
    <section className="flex flex-col gap-4">
      {dashboard.heading && (
        <header className="flex flex-col gap-1 px-4 py-3">
          <h1 className="text-lg font-semibold">{dashboard.heading}</h1>
          {dashboard.description && (
            <p className="text-sm text-muted-foreground">{dashboard.description}</p>
          )}
        </header>
      )}

      {Object.keys(dashboard.filters ?? {}).length > 0 && (
        <DashboardFilters
          filters={dashboard.filters}
          values={filterValues}
          onChange={handleFilterChange}
        />
      )}

      <DashboardGrid columns={dashboard.columns}>
        {dashboard.widgets.map((widget) => (
          <WidgetRenderer key={widget.id} widget={widget} />
        ))}
      </DashboardGrid>
    </section>
  );
}
```

- `<DashboardGrid>` may not be a render-prop wrapper; it might iterate `widgets` itself.

The exact API names for widget setters (`value()`, `description()`, `sparkline()`, `chartType()`, etc.) come from Task 0.1 step 7. Adjust the example below to the real fluent API.

Create `apps/demo/app/Arqel/Dashboards/MainDashboard.php`:

```php
<?php

declare(strict_types=1);

namespace App\Arqel\Dashboards;

use App\Models\Post;
use Arqel\Widgets\ChartWidget;
use Arqel\Widgets\Dashboard;
use Arqel\Widgets\StatWidget;
use Arqel\Widgets\TableWidget;

final class MainDashboard
{
    public static function build(): Dashboard
    {
        return Dashboard::make(id: 'main', label: 'Dashboard')
            ->heading('Overview')
            ->description('At-a-glance metrics for the demo app.')
            ->columns(3)
            ->widgets([
                self::totalPostsWidget(),
                self::postsPerDayWidget(),
                self::recentPostsWidget(),
            ]);
    }

    private static function totalPostsWidget(): StatWidget
    {
        return StatWidget::make('total_posts')
            ->heading('Total posts')
            ->columnSpan(1)
            ->value(fn (): int => Post::query()->count());
    }

    private static function postsPerDayWidget(): ChartWidget
    {
        return ChartWidget::make('posts_per_day')
            ->heading('Posts per day (last 14)')
            ->columnSpan(2)
            ->chartType('line')
            ->dataset(function (): array {
                $rows = Post::query()
                    ->selectRaw('DATE(created_at) as day, COUNT(*) as total')
                    ->where('created_at', '>=', now()->subDays(13)->startOfDay())
                    ->groupBy('day')
                    ->orderBy('day')
                    ->get();

                return [
                    'labels' => $rows->pluck('day')->all(),
                    'datasets' => [[
                        'label' => 'Posts',
                        'data' => $rows->pluck('total')->all(),
                    ]],
                ];
            });
    }

    private static function recentPostsWidget(): TableWidget
    {
        return TableWidget::make('recent_posts')
            ->heading('Recent posts')
            ->columnSpan(3)
            ->columns([
                ['name' => 'title', 'label' => 'Title'],
                ['name' => 'status', 'label' => 'Status'],
                ['name' => 'created_at', 'label' => 'Created'],
            ])
            ->query(fn () => Post::query()->latest()->limit(5));
    }
}
```

**Adapt setter names + return shapes** to whatever the real `StatWidget` / `ChartWidget` / `TableWidget` expose. The fluent calls above are illustrative — the inspection in Task 0.1 step 7 yields the correct surface.
