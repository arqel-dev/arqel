## Pre-plan finding (essential)

The spec specified `PathResolver` (URL `/admin/{tenant}/projects`). Inspection during plan-writing revealed `PathResolver::extractFirstSegment()` reads the **first** path segment, which for `/admin/{tenant}/...` is the panel prefix `admin` — placed in the `ignoreSegments` config to avoid mis-resolving — but then no tenant gets resolved at all, because the resolver does not look past index 0.

Fixing this would require either a custom resolver fork or moving the panel path to `''` (root), both invasive. After re-discussing with the user, switched to **`AuthUserResolver`** (canonical for "user with multiple tenants" scenarios):

- URL stays plain `/admin/projects` (no tenant segment)
- `user.current_tenant_id` column persists active tenant
- `<TenantSwitcher>` posts to `/admin/tenants/{id}/switch` → controller updates the user column → redirect to `/admin` → resolver re-reads from user

This is zero-config (matches Jetstream/Spark conventions out of the box) and avoids any routing fork. **All references to PathResolver in the spec are superseded by AuthUserResolver in this plan.**

- [ ] **Step 3: Create the React component**

`/home/diogo/PhpstormProjects/arqel/packages-js/ui/src/shell/TenantSwitcher.tsx`:

```typescript
/**
 * `<TenantSwitcher>` — dropdown that lists tenants the user can
 * access and dispatches `POST /admin/tenants/{id}/switch` via
 * Inertia router. Renders nothing when there is no current tenant
 * or the user only has access to one tenant.
 *
 * Mounts in the `tenantSwitcher` slot of `<Topbar>`. The current
 * tenant is read from `props.tenant.current` shared by the
 * server-side `HandleArqelInertiaRequests`; available tenants
 * come from the same payload.
 */

import type { TenantSummary } from '@arqel-dev/types/tenant';
import { router } from '@inertiajs/react';
import type { JSX } from 'react';
import { Button } from '../action/Button.js';
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from '../shadcn/ui/dropdown-menu.js';

export interface TenantSwitcherProps {
  current: TenantSummary | null;
  available: ReadonlyArray<TenantSummary>;
  /**
   * Path template for switching; defaults to
   * `/admin/tenants/{id}/switch`.
   */
  switchUrl?: (tenantId: string | number) => string;
}

const DEFAULT_SWITCH_URL = (id: string | number): string =>
  `/admin/tenants/${id}/switch`;

export function TenantSwitcher({
  current,
  available,
  switchUrl,
}: TenantSwitcherProps): JSX.Element | null {
  if (!current || available.length <= 1) return null;

  const buildUrl = switchUrl ?? DEFAULT_SWITCH_URL;
  const label = current.name ?? `Tenant ${current.id}`;

  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="ghost" size="sm" data-testid="tenant-switcher-trigger">
          {label}
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end">
        {available.map((tenant) => (
          <DropdownMenuItem
            key={String(tenant.id)}
            data-testid={`tenant-switcher-option-${String(tenant.id)}`}
            onSelect={() => {
              router.post(buildUrl(tenant.id), {}, { preserveScroll: true });
            }}
          >
            {tenant.name ?? `Tenant ${tenant.id}`}
            {tenant.id === current.id ? ' ✓' : ''}
          </DropdownMenuItem>
        ))}
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
```

`data-testid` attributes are stable hooks for the E2E spec to avoid relying on text-based queries that locales might change.

```php
<?php

declare(strict_types=1);

namespace App\Http\Middleware;

use Arqel\Core\Http\Middleware\HandleArqelInertiaRequests as BaseHandle;
use Arqel\Tenant\TenantManager;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Http\Request;

final class HandleArqelInertiaRequests extends BaseHandle
{
    /**
     * @return array<string, mixed>
     */
    public function share(Request $request): array
    {
        return array_merge(parent::share($request), [
            'tenant' => $this->tenantContext($request),
        ]);
    }

    /**
     * @return array{current: array{id: int|string, name: string|null, slug: string|null, logo: string|null}|null, available: array<int, array{id: int|string, name: string|null, slug: string|null, logo: string|null}>}
     */
    private function tenantContext(Request $request): array
    {
        $manager = app(TenantManager::class);
        $current = $manager->current();
        $user = $request->user();

        return [
            'current' => $this->serialise($current),
            'available' => $user ? array_map(
                fn (Model $tenant): array => $this->serialise($tenant) ?? [],
                $manager->availableFor($user),
            ) : [],
        ];
    }

    /**
     * @return array{id: int|string, name: string|null, slug: string|null, logo: string|null}|null
     */
    private function serialise(?Model $tenant): ?array
    {
        if ($tenant === null) {
            return null;
        }

        return [
            'id' => $tenant->getKey(),
            'name' => $tenant->name ?? null,
            'slug' => $tenant->slug ?? null,
            'logo' => $tenant->logo ?? null,
        ];
    }
}
```

```typescript
/**
 * Reset the tenant-demo database to a known state for each E2E test.
 *
 * Runs migrate:fresh --seed --force then re-runs arqel:make-user
 * (the seeder also creates the admin user idempotently and
 * attaches both tenants).
 */
export function resetDatabase(): void {
  runArtisan(['migrate:fresh', '--seed', '--force']);
  runArtisan([
    'arqel:make-user',

  // (make-user runs after migrate:fresh wipes everything; seeder is
```

```typescript
    // Tenant switcher trigger shows Acme
    const trigger = loggedInPage.locator('[data-testid="tenant-switcher-trigger"]');
    await expect(trigger).toContainText(/Acme/i);

    // Open dropdown + click Globex option
    await trigger.click();
    await loggedInPage
      .locator('[data-testid^="tenant-switcher-option-"]')
      .filter({ hasText: /Globex/i })
      .click();

    // After switch, page reloads (Inertia POST → redirect to /admin)
    await loggedInPage.waitForLoadState('networkidle');

    // Re-navigate to /admin/projects (the controller redirects to /admin
    // intended; the user lands on whatever the panel's home is)
    await loggedInPage.goto('/admin/projects');
    await expect(loggedInPage.locator('table tbody tr').first()).toBeVisible();

    // Now scoped to Globex — 5 projects, all distinct from Acme
    await expect(loggedInPage.locator('table tbody tr')).toHaveCount(5);
    const globexNames = await loggedInPage
      .locator('table tbody tr td:first-child')
      .allInnerTexts();
    expect(globexNames).toHaveLength(5);

    // Acme and Globex projects must be disjoint (factory uses unique())
    const overlap = acmeNames.filter((n) => globexNames.includes(n));
    expect(overlap).toHaveLength(0);

    // Trigger now reads Globex
    await expect(trigger).toContainText(/Globex/i);
  });
```
